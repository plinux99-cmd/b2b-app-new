project_name     = "aerowise"
environment      = "prod"
aws_region       = "us-east-1"
cluster_version  = "1.33"

# Network (creates new)
create_vpc = true
vpc_cidr   = "10.0.0.0/16"

# Database
db_name           = "aerowise_db"
db_username       = "postgres"
db_password       = "generate-strong-password-here"
db_instance_class = "db.t3.medium"
allocated_storage = 20
storage_type      = "gp3"     # ← NEW: Always uses gp3

# EKS
desired_size  = 2
min_size      = 1
max_size      = 4
instance_types = ["t3.medium"]

# API Gateway
create_authorizer = true

# CDN
create_cdn      = true
create_waf_cdn  = false
create_waf_api  = false

# Security
cloudtrail_enabled = true
guardduty_enabled  = true
